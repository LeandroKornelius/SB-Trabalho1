void leitorDeArquivo(const string& nome_do_arquivo)
	*lê o arquivo de texto linha por linha
	-chama analisadorSintatico
		-passa uma string de linha
	

vector<string> analisadorSintatico(const string& linha)
	*verifica se uma linha é sintaticamente correta
	-chama percorreLinha
		-passa uma string de linha
	-chama verificaTokens
		-passa retorno de percorreLinha
	-chama inSyntax
		-passa retorno de verificaTokens

vector<string> percorreLinha(const string& str)
	*separa a string de linha em um vector de palavras
	-exemplo:
		"ADD N1 N2" -> {"ADD", "N1", "N2"}
		"FRONT: LOAD N3 -> {"FRONT", "LOAD", "N3"}

vector<int> verificaTokens(const vector<string>& v)
	*verifica o tipo de cada palavra
	-chama analisadorLexico
	// 1..15 = Palavras Reservadas
	// 20 = Rótulo
	// 30 = Constante
	-exemplo:
		{"ADD", "N1", "N2"} -> {1, 20, 20}

int analisadorLexico(const string& str, int posicaoDePalavra)
	*dado uma string verifica seu tipo
	*também realiza criação da tabela de símbolos e da lista de pendências
	-exemplo:
		"ADD" -> 1
		"N1" -> 20
		"3" -> 30

bool inSyntax(const vector<int>& v)
	*verifica se o vector<int> está na tabela de sintaxe da linguagem
